git init ==> git initialisation 

git status ==> list untracked files

git add -A ==> adding all files to staging area 

git commit -m "message" ==> moving staged files to local register 

git clone ==> exact copy of remote repository to local space 

git push ==> local space to remote repository 

git pull ==> remote repository to local space 

git remote add name remote_path

git remote -v ==> list all remote name
